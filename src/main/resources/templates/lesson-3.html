2. ContactForm DTO (Validation Annotations):

public class ContactForm {
    @NotBlank(message = "Nama wajib diisi")
    @Size(min = 3, message = "Nama minimal 3 karakter")
    private String name;

    @NotBlank(message = "Email wajib diisi")
    @Email(message = "Format email tidak valid")
    private String email;

    @NotBlank(message = "Subjek wajib diisi")
    private String subject;

    @NotBlank(message = "Pesan wajib diisi")
    @Size(min = 10, message = "Pesan minimal 10 karakter")
    private String message;
}

3. Spring Controller (Backend):

@PostMapping("/contact")
public String submitContact(@Valid ContactForm form,
                           BindingResult bindingResult,
                           Model model) {
    if (bindingResult.hasErrors()) {
        // Convert errors ke Map
        Map<String, String> errors = new HashMap<>();
        bindingResult.getFieldErrors().forEach(error ->
            errors.put(error.getField(), error.getDefaultMessage())
        );
        model.addAttribute("errors", errors);
        model.addAttribute("form", form);
        return "fragments/contact-form";
    }

    model.addAttribute("name", form.getName());
    return "fragments/contact-success";
}
